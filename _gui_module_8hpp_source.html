202 void ExecCommand(const char* command_line) {
203 AddLog("> %s\n", command_line);
204 // Insert into history. First find match and delete it so it can be pushed to the back.
205 // This isn't trying to be smart or optimal.
206 HistoryPos = -1;
207 for (int i = History.Size - 1; i >= 0; i--)
208 if (Stricmp(History[i], command_line) == 0) {
209 free(History[i]);
210 History.erase(History.begin() + i);
211 break;
212 }
213 History.push_back(Strdup(command_line));
214 // Process command
215 if (Stricmp(command_line, "CLEAR") == 0) {
216 ClearLog();
217 } else if (Stricmp(command_line, "HELP") == 0) {
218 AddLog("Commands:");
219 for (int i = 0; i < Commands.Size; i++) AddLog("- %s", Commands[i]);
220 } else if (Stricmp(command_line, "HISTORY") == 0) {
221 int first = History.Size - 10;
222 for (int i = first > 0 ? first : 0; i < History.Size; i++) AddLog("%3d: %s\n", i, History[i]);
223 } else {
224 AddLog("Unknown command: '%s'\n", command_line);
225 }
226 // On command input, we scroll to bottom even if AutoScroll==false
227 ScrollToBottom = true;
228 }

235 int TextEditCallback(ImGuiInputTextCallbackData* data) {
236 switch (data->EventFlag) {
237 case ImGuiInputTextFlags_CallbackCompletion: {
238 // Example of TEXT COMPLETION - Locate beginning of current word
239 const char* word_end = data->Buf + data->CursorPos;
240 const char* word_start = word_end;
241 while (word_start > data->Buf) {
242 const char c = word_start[-1];
243 if (c == ' ' || c == '\t' || c == ',' || c == ';') break;
244 word_start--;
245 }
246 // Build a list of candidates
247 ImVector<const char*> candidates;
248 for (int i = 0; i < Commands.Size; i++)
249 if (Strnicmp(Commands[i], word_start, (int)(word_end - word_start)) == 0) candidates.push_back(Commands[i]);
250 if (candidates.Size == 0) {
251 // No match
252 AddLog("No match for \"%.*s\"!\n", (int)(word_end - word_start), word_start);
253 } else if (candidates.Size == 1) {
254 // Single match. Delete the beginning of the word and replace it entirely so we've got nice casing.
255 data->DeleteChars((int)(word_start - data->Buf), (int)(word_end - word_start));
256 data->InsertChars(data->CursorPos, candidates[0]);
257 data->InsertChars(data->CursorPos, " ");
258 } else {
259 // Multiple matches. Complete as much as we can..
260 // So inputing "C"+Tab will complete to "CL" then display "CLEAR" and "CLASSIFY" as matches.
261 int match_len = (int)(word_end - word_start);
262 for (;;) {
263 int c = 0;
264 bool all_candidates_matches = true;
265 for (int i = 0; i < candidates.Size && all_candidates_matches; i++)
266 if (i == 0) c = toupper(candidates[i][match_len]);
267 else if (c == 0 || c != toupper(candidates[i][match_len]))
268 all_candidates_matches = false;
269 if (!all_candidates_matches) break;
270 match_len++;
271 }
272 if (match_len > 0) {
273 data->DeleteChars((int)(word_start - data->Buf), (int)(word_end - word_start));
274 data->InsertChars(data->CursorPos, candidates[0], candidates[0] + match_len);
275 }
276 // List matches
277 AddLog("Possible matches:\n");
278 for (int i = 0; i < candidates.Size; i++) AddLog("- %s\n", candidates[i]);
279 }
280 break;
281 }
282 case ImGuiInputTextFlags_CallbackHistory: {
283 // Example of HISTORY
284 const int prev_history_pos = HistoryPos;
285 if (data->EventKey == ImGuiKey_UpArrow) {
286 if (HistoryPos == -1) HistoryPos = History.Size - 1;
287 else if (HistoryPos > 0)
288 HistoryPos--;
289 } else if (data->EventKey == ImGuiKey_DownArrow) {
290 if (HistoryPos != -1)
291 if (++HistoryPos >= History.Size) HistoryPos = -1;
292 }
293 // A better implementation would preserve the data on the current input line along with cursor position.
294 if (prev_history_pos != HistoryPos) {
295 const char* history_str = (HistoryPos >= 0) ? History[HistoryPos] : "";
296 data->DeleteChars(0, data->BufTextLen);
297 data->InsertChars(0, history_str);
298 }
299 }
300 }
301 return 0;
302 }
